  <script src="https://unpkg.com/react@17/umd/react.development.js"></script>
  <script src="https://unpkg.com/react-dom@17/umd/react-dom.development.js"></script>
  <script src="https://unpkg.com/babel-standalone@6/babel.min.js"></script>

  <script type="text/babel">
    function App() {
      const [userChoice, setUserChoice] = React.useState("");
      const [computerChoice, setComputerChoice] = React.useState("");
      const [result, setResult] = React.useState("");

      const choices = ["Rock ✊", "Paper ✋", "Scissors✌️"];

      const playGame = (choice) => {
        setUserChoice(choice);
        const compChoice = choices[Math.floor(Math.random() * 3)];
        setComputerChoice(compChoice);

        const user = choice.split(" ")[0].toLowerCase();
        const comp = compChoice.split(" ")[0].toLowerCase();

        if (user === comp) {
          setResult("It's a tie! 😶");
        } else if (
          (user === "rock" && comp === "scissors") ||
          (user === "scissors" && comp === "paper") ||
          (user === "paper" && comp === "rock")
        ) {
          setResult("You win! 💖👑");
        } else {
          setResult("You lose! 😭💔");
        }
      };

      return (
        <div>
          {choices.map((choice, index) => (
            <button key={index} onClick={() => playGame(choice)}>
              {choice}
            </button>
          ))}
          <div className="result">
            <p><strong>You:</strong> {userChoice}</p>
            <p><strong>Computer:</strong> {computerChoice}</p>
            <h3>{result}</h3>
          </div>
        </div>
      );
    }

    ReactDOM.render(<App />, document.getElementById("app"));
  </script>
